endpaint

Use the number keys 1-8 to create brush strokes that look 3D.
Use the scrollbar while holding "shift" to change the size of the 3D strokes.
(holding "shift" simply prevents the whole page from moving up and down)
Keys 9 & 0 allow for a smooth solid brush stroke in black or white.
press "P" to save your work and "X" to reset the canvas.

let pg;
var initials ='endpaint';
var screenbgR = 204;

var choice = '1'; // starting choice, so it is not empty
var lastscreenshot=61; // last screenshot never taken

function setup() {
createCanvas(1280, 720);  // canvas size
pg = createGraphics (1280, 720);
pg.background(screenbgR, screenbgG, screenbgB);
}

function draw() {
  image(pg,0,0);
  if (keyIsPressed) {
    if (key == 1||key == 2||key == 3||key == 4||key == 5||key == 6||key == 7||key == 8||key == 9||key == 0){
    choice = key; // set choice to the key that was pressed
    }
    clear_print(); // check to see if git is clear screen or save image
    
  }
  if (mouseIsPressed){
    newkeyChoice(choice);  // if the mouse is pressed call newkeyChoice
  }

  if(choice == '1'){ 
   stroke(255);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '2'){ 
   stroke(0);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '3'){ 
   stroke(255,255,0);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '4'){ 
   stroke(0,0,255);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '5'){ 
   stroke(255,128,0);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '6'){ 
   stroke(0,255,0);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '7'){ 
   stroke(0,128,0);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '8'){ 
   stroke(0,128,128);
   strokeWeight(6);
   fill(0,0);
   ellipse(mouseX,mouseY,bsize);
  }
  if(choice == '9'){ 
   stroke(255);
   strokeWeight(2);
   fill(0,0);
   ellipse(mouseX,mouseY,12);
  }
  if(choice == '0'){ 
   stroke(0);
   strokeWeight(2);
   fill(0,0);
   ellipse(mouseX,mouseY,12);
  }
}
function mouseWheel(scroll) {
   bsize+=scroll.delta/8; // sets brush size depending on scroll wheel interaction

}
function newkeyChoice(toolChoice) { //toolchoice is the key that was pressed

 if (toolChoice == '1' ) {  // white sphere
    pg.image(ball1, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);   
   
  } else if (toolChoice == '2') { // black sphere
    pg.image(ball2, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '3') { // yellow sphere
    pg.image(ball3, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '4') { // blue sphere
    pg.image(ball4, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '5') { // orange sphere
    pg.image(ball5, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '6') { // lime green sphere
    pg.image(ball6, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '7') { // dark green sphere
    pg.image(ball7, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '8') { // dark cyan sphere
    pg.image(ball8, mouseX-(round(bsize/2)), mouseY-(round(bsize/2)),bsize,bsize);
    
  } else if (toolChoice == '9') { // white stroke tool
    pg.stroke(255);
    pg.strokeWeight(12);
    pg.line(mouseX, mouseY, pmouseX, pmouseY);
    
  } else if (toolChoice == '0') { // black stroke tool
    pg.stroke(0);
    pg.strokeWeight(12);
    pg.line(mouseX, mouseY, pmouseX, pmouseY);
    
  }
 }


function clear_print() {
// this will do one of two things, x clears the screen by resetting the background
// p calls the routine saveme, which saves a copy of the screen
  if (key == 'x' || key == 'X') {
    pg.background(screenbgR, screenbgG, screenbgB); // set the screen back to the background color
  } else if (key == 'p' || key == 'P') {
     saveme();  // call saveme which saves an image of the screen
  }
}

function saveme(){
    //this will save the name as the intials, date, time and a millis counting number.
    // it will always be larger in value then the last one.
  filename=initials+day() + hour() + minute() +second();
  if (second()!=lastscreenshot) { // don't take a screenshot if you just took one
    saveCanvas(filename, 'png');
  }
  lastscreenshot=second(); // set this to the current second so no more than one per second
  
}
